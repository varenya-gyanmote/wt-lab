import java.io.*;
import javax.servlet.*;
import javax.servlet.http.*;

public class CalculatorServlet extends HttpServlet {
    public void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        response.setContentType("text/html");
        PrintWriter out = response.getWriter();
        
        int num1 = Integer.parseInt(request.getParameter("n1"));
        int num2 = Integer.parseInt(request.getParameter("n2"));
        String op = request.getParameter("op");
        int result = 0;
        
        if(op.equals("+"))
            result = num1 + num2;
        else if(op.equals("-"))
            result = num1 - num2;
        else if(op.equals("*"))
            result = num1 * num2;
        else if(op.equals("/"))
            result = num1 / num2;
        else if(op.equals("%"))
            result = num1 % num2;
        
        out.println("<html>");
        out.println("<head><title>Calculator Result</title></head>");
        out.println("<body>");
        out.println("<center>");
        out.println("<h1>Result: " + result + "</h1>");
        out.println("</center>");
        out.println("</body>");
        out.println("</html>");
    }
}

import java.io.*;
import javax.servlet.*;
import javax.servlet.http.*;
import java.sql.*;

public class LoginServlet extends HttpServlet {
    public void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        response.setContentType("text/html");
        PrintWriter out = response.getWriter();
        
        String uname = request.getParameter("uname");
        String pass= request.getParameter("pass");
        
        try {
            Class.forName("com.mysql.jdbc.Driver");
            Connection con = DriverManager.getConnection("jdbc:mysql://localhost:3306/1294", "root", "");
            Statement stmt = con.createStatement();
            String sql = "SELECT * FROM login WHERE uname='" + uname + "' AND pass='" + pass + "'";
            ResultSet rs = stmt.executeQuery(sql);
            
            if (rs.next()) {
                out.println("Hello!! " + uname);
            } else {
                out.println("Wrong username or password");
            }
            
            con.close();
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}

	Class.forName("com.mysql.jdbc.Driver");

function loadDoc() {
var xhttp = new XMLHttpRequest();
xhttp.onreadystatechange = function() {
if (this.readyState == 4 && this.status == 200) {
myFunction(this);
}
};
xhttp.open("GET", "user.xml", true);
xhttp.send();
}
function myFunction(xml) {
var username = document.getElementById("username").value;
var password = document.getElementById("password").value;
var i;
var xmlDoc = xml.responseXML;
var x = xmlDoc.getElementsByTagName("user");
var f = false;
for (i = 0; i < x.length; i++) { 
var name = x[i].getElementsByTagName("name")[0].childNodes[0].nodeValue;
var pwd = x[i].getElementsByTagName("pwd")[0].childNodes[0].nodeValue ;
if (username === name && password === pwd) {
document.getElementById("lgn").innerHTML = name+"    "+pwd+"<br><b>Login Successful!!</b>";
f = true;
break;
}}
if (!f) {
    document.getElementById("lgn").innerHTML = "<b>Login Failure!!</b>";}}

class EmployeeHandler extends DefaultHandler {
private String s;
private boolean isTargetEmployee = false;
private boolean isFirstName = false;
private boolean isLastName = false;
private boolean isNickName = false;


private boolean isSalary = false;
public EmployeeHandler(String s) {
this.s = s;
}
public void startElement(String uri, String localName, String qName, Attributes attributes) throws
SAXException {
if (qName.equalsIgnoreCase("employee")) {
String id = attributes.getValue("id");
if (id.equals(s)) {
isTargetEmployee = true;
System.out.println("Employee ID : " + id);
}
} else if (isTargetEmployee && qName.equalsIgnoreCase("firstname")) {
isFirstName = true;
} else if (isTargetEmployee && qName.equalsIgnoreCase("lastname")) {
isLastName = true;
} else if (isTargetEmployee && qName.equalsIgnoreCase("nickname")) {
isNickName = true;
} else if (isTargetEmployee && qName.equalsIgnoreCase("salary")) {
isSalary = true;
}



}
public void endElement(String uri, String localName, String qName) throws SAXException {
if (qName.equalsIgnoreCase("employee")) {
if (isTargetEmployee) {
System.out.println("End Element :" + qName);
}
isTargetEmployee = false;
}
}

}
}

myFunction(this);
